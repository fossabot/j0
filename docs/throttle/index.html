import {setTimeout} from '..';
import isUndefined from '../isUndefined';

function throttle(fn, interval = 0, context) {
	let lastArgs = [];
	let timer = null;
	let scheduled = false;
	function call(...args) {
		const thisArg = isUndefined(context) ? this : context;
		lastArgs = args;
		if (timer) {
			scheduled = true;
		} else {
			fn.apply(thisArg, args);
			timer = setTimeout(() => {
				timer = null;
				if (scheduled) {
					scheduled = null;
					call.apply(thisArg, lastArgs);
				}
			}, interval);
		}
	}
	return call;
}
export default throttle;

import throttle from '..';

describe('throttle', function () {

	it('should call the function intermittently', () => {
		return new Promise((resolve) => {
			let count = 0;
			const execute = throttle((value) => {
				count += value;
				if (1 < count) {
					resolve(count);
				}
			}, 50);
			execute(1);
			execute(2);
			execute(3);
			execute(4);
			execute(5);
		}).then((result) => {
			assert.equal(result, 6);
		});
	});

});
