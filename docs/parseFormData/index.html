import trim from '../String/trim';
import forEach from '../Array/forEach';
import {
	decodeURIComponent,
	FormData
} from 'j0';
function parse(body, form = new FormData()) {
	forEach(trim(body).split('&'), function (data) {
		if (data) {
			let [name, ...parts] = data.split('=');
			name = decodeURIComponent(name.replace(/\+/g, ' '));
			parts = decodeURIComponent(parts.join('=').replace(/\+/g, ' '));
			form.append(name, parts);
		}
	});
	return form;
}
export default parse;

import parse from '..';

describe('FormData/parse', function () {

	it('should parse string', function () {
		const form = parse('a=b&c=d', {
			data: {},
			append: function (key, value) {
				this.data[key] = value;
			}
		});
		assert.deepEqual(form.data, {
			a: 'b',
			c: 'd'
		});
	});

});
