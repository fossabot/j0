Checks whether every arguments is an array or not.

@param {Number} args

Source

const {isArray: _isArray} = Array;
function isArray(...args) {
	return args.every(_isArray);
}

export default isArray;

Tests

import assert from 'assert';
import isArray from '.';

describe('isArray', function () {
	it('should return true is the argument is an number', function () {
		assert.equal(isArray(0), false);
	});
	it('should return true is the argument is an array', function () {
		assert.equal(isArray([]), true);
	});
});
